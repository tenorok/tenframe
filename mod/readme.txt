Структура модуля:

app/												// Директория программных файлов
	controller/
		mod_{module-name}_{controller-name}.php		// Правило наименования контроллеров
	model/
		mod_{module-name}_m_{model-name}.php		// Правило наименования моделей
view/												// Представления. Структура в традиционном для tenframe виде
conf/												// Директория для хранения конфигураций
init.php											// Инициализатор модуля, подключается автоматически, если модуль инициализирован
readme.md											// Описание модуля отображается на странице: /mod/{mod-name}/

Database:
	tmod_{module-name}								// Правило наименования БД
	tmod_{module-name}_{table}						// Правило наименования таблиц в БД
